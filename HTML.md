```html
<!-- Introduction to HTML -->

<!-- HTML (HyperText Markup Language) is the standard language for creating web pages and applications. -->

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Intro to HTML</title>
</head>
<body>
    <!-- 1. Headings -->
    <h1>Main Heading</h1>
    <h2>Subheading</h2>
    <h3>Smaller Subheading</h3>

    <!-- 2. Paragraphs -->
    <p>This is a paragraph. HTML allows you to structure content using various tags.</p>

    <!-- 3. Links -->
    <a href="https://www.example.com" target="_blank">Visit Example</a>

    <!-- 4. Images -->
    <img src="https://via.placeholder.com/150" alt="Placeholder Image">

    <!-- 5. Lists -->
    <!-- Ordered List -->
    <ol>
        <li>First item</li>
        <li>Second item</li>
        <li>Third item</li>
    </ol>

    <!-- Unordered List -->
    <ul>
        <li>Bullet point 1</li>
        <li>Bullet point 2</li>
        <li>Bullet point 3</li>
    </ul>

    <!-- 6. Tables -->
    <table border="1">
        <tr>
            <th>Column 1</th>
            <th>Column 2</th>
        </tr>
        <tr>
            <td>Row 1, Cell 1</td>
            <td>Row 1, Cell 2</td>
        </tr>
        <tr>
            <td>Row 2, Cell 1</td>
            <td>Row 2, Cell 2</td>
        </tr>
    </table>

    <!-- 7. Forms -->
    <form action="/submit" method="POST">
        <label for="name">Name:</label>
        <input type="text" id="name" name="name" required>

        <label for="email">Email:</label>
        <input type="email" id="email" name="email" required>

        <button type="submit">Submit</button>
    </form>

    <!-- 8. Divs and Spans -->
    <div style="border: 1px solid black; padding: 10px;">
        This is a div element, often used for layout and styling.
    </div>

    <span style="color: red;">This is a span element, often used for inline styling.</span>

</body>
</html>
```html
